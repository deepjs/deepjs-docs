Intro

deep.collider provides way to manage fine grained collision when applying two layer together.

.replace()

deep({ b:[1,2,3] })
.up({
    b:deep.collider.replace([4,5])
})
.equal({ b:[4,5] })
.log()

.array.insertAt()

deep({ b:[1,2,3] })
.up({
    b:deep.collider["array.insertAt"]([4,5],2)
})
.equal({ b:[1,2,4,5,3] })
.log()

Custom collider

deep.collider.Collider.add("validate", function(input, schema){
	var report = deep.validate(input, schema);
	if(!report.valid)
		throw deep.errors.PreconditionFail(report);
		else deep.log("valid at collision.")
	return input;
});
